package com.lisa.service.report;

import com.lisa.enums.ReportType;
import lombok.RequiredArgsConstructor;
import org.springframework.stereotype.Component;

@RequiredArgsConstructor
@Component
public class GenerateReportInstanceFactory {

    private final GenerateLeadStatusReportService generateLeadStatusReportService;
    private final GenerateOptOutReportService generateOptOutReportService;
    private final GenerateAppointmentActivityReportService generateAppointmentActivityReportService;
    private final GenerateResponseInsightReportService generateResponseInsigntReportService;
    private final GenerateInboundAppointmentActivityReportService generateInboundAppointmentActivityReportService;
    private final GeneratePerformanceSummaryReportService generatePerformanceSummaryReportService;
    private final GeneratePerformanceSummaryListReportService generatePerformanceSummaryListReportService;
    private final GenerateControlGroupTreatmentService generateControlGroupTreatmentService;
    private final GenerateLeadStatusActivityReportService generateLeadStatusActivityReportService;
    private final GenerateMBConnectedCarNotTextableNRAAService generateMBConnectedCarNotTextableNRAAService;
    private final GenerateInclusionReportService generateInclusionReportService;
    private final GenerateInclusionPlannedReportService generateInclusionPlannedReportService;
    private final GenerateExclusionReportService generateExclusionReportService;
    private final GenerateAppointmentStatusReportService appointmentStatusService;

    public GenerateReportService getInstanceByType(ReportType reportType) {
        switch (reportType) {
            case LEAD_STATUS:
                return generateLeadStatusReportService;
            case OPT_OUT:
                return generateOptOutReportService;
            case APPT_ACTIVITY:
                return generateAppointmentActivityReportService;
            case RESPONSE_INSIGHT:
                return generateResponseInsigntReportService;
            case INBOUND_ACTIVITY_REPORT:
                return generateInboundAppointmentActivityReportService;
            case PERFORMANCE_SUMMARY:
                return generatePerformanceSummaryReportService;
            case PERFORMANCE_SUMMARY_PER_STORE:
                return generatePerformanceSummaryListReportService;
            case CONTROL_GROUP_TREATMENT:
                return generateControlGroupTreatmentService;
            case LEAD_STATUS_ACTIVITY:
                return generateLeadStatusActivityReportService;
            case MB_CONNECTEDCAR_NOTTEXTTABLE_NRAA:
                return generateMBConnectedCarNotTextableNRAAService;
            case AUTONATION_EXCLUSION_REPORT:
                return generateExclusionReportService;
            case AUTONATION_INCLUSION_REPORT:
                return generateInclusionReportService;
            case AUTONATION_INCLUSION_PLANNED_DATE_REPORT:
                return generateInclusionPlannedReportService;
            case AFTER_HOURS_DETAIL:
            case NO_SERVICE_FULL:
            case NO_SERVICE_PARTIAL:
                return appointmentStatusService;
            default:
                throw new IllegalArgumentException("Unknown report type : " + reportType);
        }
    }
}
